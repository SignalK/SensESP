41// TOFU verification callback - called during SSL handshake
42// Returns 0 to allow connection, non-zero to reject
43static int tofu_verify_callback(void* ctx, mbedtls_x509_crt* crt,
44 int depth, uint32_t* flags) {
45 // Only check the server certificate (depth 0), not the CA chain
46 if (depth != 0) {
47 *flags = 0; // Clear errors for intermediate certs
48 return 0;
49 }
50
51 SKWSClient* client = static_cast<SKWSClient*>(ctx);
52 if (client == nullptr) {
53 ESP_LOGW("SKWSClient", "TOFU: No client context, allowing connection");
54 *flags = 0;
55 return 0;
56 }
57
58 // Compute SHA256 of the certificate
59 uint8_t sha256[32];
60 mbedtls_sha256_context sha256_ctx;
61 mbedtls_sha256_init(&sha256_ctx);
62 mbedtls_sha256_starts(&sha256_ctx, 0); // 0 = SHA256 (not SHA224)
63 mbedtls_sha256_update(&sha256_ctx, crt->raw.p, crt->raw.len);
64 mbedtls_sha256_finish(&sha256_ctx, sha256);
65 mbedtls_sha256_free(&sha256_ctx);
66
67 char hex[65];
68 sha256_to_hex(sha256, hex);
69 String current_fingerprint = String(hex);
70
71 ESP_LOGD("SKWSClient", "Server certificate fingerprint: %s", hex);
72
73 if (!client->is_tofu_enabled()) {
74 // TOFU disabled, allow any certificate
75 ESP_LOGD("SKWSClient", "TOFU disabled, allowing connection");
76 *flags = 0;
77 return 0;
78 }
79
80 if (!client->has_tofu_fingerprint()) {
81 // First connection - capture the fingerprint
82 ESP_LOGI("SKWSClient", "TOFU: First connection, capturing fingerprint: %s", hex);
83 client->set_tofu_fingerprint(current_fingerprint);
84 *flags = 0;
85 return 0;
86 }
87
88 // Verify against stored fingerprint
89 if (client->get_tofu_fingerprint() == current_fingerprint) {
90 ESP_LOGD("SKWSClient", "TOFU: Fingerprint verified successfully");
91 *flags = 0;
92 return 0;
93 }
94
95 // Fingerprint mismatch!
96 ESP_LOGE("SKWSClient", "TOFU: Fingerprint mismatch!");
97 ESP_LOGE("SKWSClient", " Expected: %s", client->get_tofu_fingerprint().c_str());
98 ESP_LOGE("SKWSClient", " Received: %s", hex);
99 // Return error to reject the connection
100 return MBEDTLS_ERR_X509_CERT_VERIFY_FAILED;
101}

103// Certificate bundle attach function for TOFU verification
104// Sets up a custom verification callback that implements Trust On First Use
105static esp_err_t tofu_crt_bundle_attach(void* conf) {
106 mbedtls_ssl_config* ssl_conf = static_cast<mbedtls_ssl_config*>(conf);
107 // Use OPTIONAL so we can handle verification ourselves
108 mbedtls_ssl_conf_authmode(ssl_conf, MBEDTLS_SSL_VERIFY_OPTIONAL);
109 mbedtls_ssl_conf_verify(ssl_conf, tofu_verify_callback, ws_client);
110 ESP_LOGD("SKWSClient", "TOFU verification callback installed");
111 return ESP_OK;
112}

358void SKWSClient::process_received_updates() {
359 SKListener::take_semaphore();
360
361 const std::vector<SKListener*>& listeners = SKListener::get_listeners();
362 const std::vector<SKPutListener*>& put_listeners =
363 SKPutListener::get_listeners();
364
365 take_received_updates_semaphore();
366 int num_updates = received_updates_.size();
367 while (!received_updates_.empty()) {
368 JsonDocument& doc = received_updates_.front();
369
370 const char* path = doc["path"];
371 JsonObject value = doc.as<JsonObject>();
372
373 for (size_t i = 0; i < listeners.size(); i++) {
374 SKListener* listener = listeners[i];
375 if (listener->get_sk_path().equals(path)) {
376 listener->parse_value(value);
377 }
378 }
379 // to be able to parse values of Put Listeners GA
380 for (size_t i = 0; i < put_listeners.size(); i++) {
381 SKPutListener* listener = put_listeners[i];
382 if (listener->get_sk_path().equals(path)) {
383 listener->parse_value(value);
384 }
385 }
386 received_updates_.pop_front();
387 }
388 release_received_updates_semaphore();
389 delta_rx_count_producer_.set(num_updates);
390
391 SKListener::release_semaphore();
392}

401void SKWSClient::on_receive_put(JsonDocument& message) {
402 // Process PUT requests...
403 JsonArray puts = message["put"];
404 size_t response_count = 0;
405 for (size_t i = 0; i < puts.size(); i++) {
406 JsonObject value = puts[i];
407 const char* path = value["path"];
408 String str_val = value["value"].as<String>();
409
410 SKListener::take_semaphore();
411 const std::vector<SKPutListener*>& listeners =
412 SKPutListener::get_listeners();
413 for (size_t i = 0; i < listeners.size(); i++) {
414 SKPutListener* listener = listeners[i];
415 if (listener->get_sk_path().equals(path)) {
416 take_received_updates_semaphore();
417 received_updates_.push_back(value);
418 release_received_updates_semaphore();
419 response_count++;
420 }
421 }
422 SKListener::release_semaphore();
423
424 // Send back a request response...
425 JsonDocument put_response;
426 put_response["requestId"] = message["requestId"];
427 if (response_count == puts.size()) {
428 // We found a response for every PUT request
429 put_response["state"] = "COMPLETED";
430 put_response["statusCode"] = 200;
431 } else {
432 // One or more requests did not have a matching path
433 put_response["state"] = "FAILED";
434 put_response["statusCode"] = 405;
435 }
436 String response_text;
437 serializeJson(put_response, response_text);
438 esp_websocket_client_send_text(client_, response_text.c_str(),
439 response_text.length(), portMAX_DELAY);
440 }

457bool SKWSClient::get_mdns_service(String& server_address,
458 uint16_t& server_port) {
459 // get IP address using an mDNS query
460 // Try SSL service first, then fall back to non-SSL
461 int num = MDNS.queryService("signalk-wss", "tcp");
462 if (num > 0) {
463 // Found SSL-enabled server
464 ssl_enabled_ = true;
465 ESP_LOGI(__FILENAME__, "Found Signal K server via mDNS (signalk-wss)");
466 } else {
467 // Try non-SSL service
468 num = MDNS.queryService("signalk-ws", "tcp");
469 if (num == 0) {
470 // no service found
471 return false;
472 }
473 // Found non-SSL server, disable SSL
474 ssl_enabled_ = false;
475 ESP_LOGI(__FILENAME__, "Found Signal K server via mDNS (signalk-ws)");
476 }
477
478#if ESP_ARDUINO_VERSION_MAJOR < 3
479 server_address = MDNS.IP(0).toString();
480#else
481 server_address = MDNS.address(0).toString();
482#endif
483 server_port = MDNS.port(0);
484 ESP_LOGI(__FILENAME__, "Found server %s (port %d)", server_address.c_str(),
485 server_port);
486 return true;
487}

489bool SKWSClient::detect_ssl() {
490 // Try to detect if the server requires SSL by checking for HTTP->HTTPS
491 // redirects
492 String url =
493 String("http://") + server_address_ + ":" + server_port_ + "/signalk";
494
495 ESP_LOGD(__FILENAME__, "Probing for SSL redirect at %s", url.c_str());
496
497 esp_http_client_config_t config = {};
498 config.url = url.c_str();
499 config.disable_auto_redirect = true;
500 config.timeout_ms = 10000;
501
502 esp_http_client_handle_t client = esp_http_client_init(&config);
503 if (client == nullptr) {
504 ESP_LOGE(__FILENAME__, "Failed to initialize HTTP client");
505 return false;
506 }
507
508 esp_err_t err = esp_http_client_perform(client);
509 if (err != ESP_OK) {
510 ESP_LOGD(__FILENAME__, "HTTP request failed: %s", esp_err_to_name(err));
511 esp_http_client_cleanup(client);
512 return false;
513 }
514
515 int http_code = esp_http_client_get_status_code(client);
516
517 if (http_code == 301 || http_code == 302 || http_code == 307 ||
518 http_code == 308) {
519 // Check Location header for HTTPS redirect
520 char* location = nullptr;
521 esp_http_client_get_header(client, "Location", &location);
522 esp_http_client_cleanup(client);
523
524 if (location != nullptr && strncmp(location, "https://", 8) == 0) {
525 ESP_LOGI(__FILENAME__, "SSL redirect detected to %s, enabling HTTPS/WSS",
526 location);
527 ssl_enabled_ = true;
528 save();
529 return true;
530 }
531 }
532
533 esp_http_client_cleanup(client);
534 return false;
535}

538void SKWSClient::connect() {
539 if (get_connection_state() != SKWSConnectionState::kSKWSDisconnected) {
540 return;
541 }
542
543 if (!WiFi.isConnected() && WiFi.getMode() != WIFI_MODE_AP) {
544 ESP_LOGI(
545 __FILENAME__,
546 "WiFi is disconnected. SignalK client connection will be initiated "
547 "when WiFi is connected.");
548 return;
549 }
550
551 ESP_LOGI(__FILENAME__, "Initiating websocket connection with server...");
552
553 set_connection_state(SKWSConnectionState::kSKWSAuthorizing);
554 if (use_mdns_) {
555 if (!get_mdns_service(this->server_address_, this->server_port_)) {
556 ESP_LOGE(__FILENAME__,
557 "No Signal K server found in network when using mDNS service!");
558 } else {
559 ESP_LOGI(__FILENAME__,
560 "Signal K server has been found at address %s:%d by mDNS.",
561 this->server_address_.c_str(), this->server_port_);
562 }
563 } else {
564 this->server_address_ = this->conf_server_address_;
565 this->server_port_ = this->conf_server_port_;
566 }
567
568 if (!this->server_address_.isEmpty() && this->server_port_ > 0) {
569 ESP_LOGD(__FILENAME__,
570 "Websocket is connecting to Signal K server on address %s:%d",
571 this->server_address_.c_str(), this->server_port_);
572
573 // Detect if server requires SSL (check for HTTP->HTTPS redirects)
574 if (!ssl_enabled_) {
575 detect_ssl();
576 }
577 } else {
578 // host and port not defined - don't try to connect
579 ESP_LOGD(__FILENAME__,
580 "Websocket is not connecting to Signal K server because host and "
581 "port are not defined.");
582 set_connection_state(SKWSConnectionState::kSKWSDisconnected);
583 return;
584 }
585
586 if (this->polling_href_ != "") {
587 // existing pending request
588 this->poll_access_request(this->server_address_, this->server_port_,
589 this->polling_href_);
590 return;
591 }
592
593 if (this->auth_token_ == NULL_AUTH_TOKEN) {
594 // initiate HTTP authentication
595 ESP_LOGD(__FILENAME__, "No prior authorization token present.");
596 this->send_access_request(this->server_address_, this->server_port_);
597 return;
598 }
599
600 // Test the validity of the authorization token
601 this->test_token(this->server_address_, this->server_port_);
602}

608void SKWSClient::test_token(const String server_address,
609 const uint16_t server_port) {
610 String protocol = ssl_enabled_ ? "https://" : "http://";
611 String url = protocol + server_address + ":" + server_port +
612 "/signalk/v1/stream";
613 ESP_LOGD(__FILENAME__, "Testing token with url %s", url.c_str());
614
615 const String full_token = String("Bearer ") + auth_token_;
616 ESP_LOGD(__FILENAME__, "Authorization: %s", full_token.c_str());
617
618 esp_http_client_config_t config = {};
619 config.url = url.c_str();
620 config.timeout_ms = 10000;
621#ifdef SENSESP_SSL_SUPPORT
622 if (ssl_enabled_) {
623 config.crt_bundle_attach = tofu_crt_bundle_attach;
624 config.skip_cert_common_name_check = true;
625 }
626#endif
627
628 esp_http_client_handle_t client = esp_http_client_init(&config);
629 if (client == nullptr) {
630 ESP_LOGE(__FILENAME__, "Failed to initialize HTTP client");
631 set_connection_state(SKWSConnectionState::kSKWSDisconnected);
632 return;
633 }
634
635 esp_http_client_set_header(client, "Authorization", full_token.c_str());
636
637 esp_err_t err = esp_http_client_perform(client);
638 int http_code = esp_http_client_get_status_code(client);
639 int content_length = esp_http_client_get_content_length(client);
640
641 String payload;
642 if (err == ESP_OK && content_length > 0 && content_length < 1024) {
643 char* buffer = new char[content_length + 1];
644 esp_http_client_read(client, buffer, content_length);
645 buffer[content_length] = '\0';
646 payload = String(buffer);
647 delete[] buffer;
648 }
649
650 esp_http_client_cleanup(client);
651
652 if (err == ESP_OK) {
653 ESP_LOGD(__FILENAME__, "Testing resulted in http status %d", http_code);
654 if (payload.length() > 0) {
655 ESP_LOGD(__FILENAME__,
656 "Returned payload (length %d) is: ", payload.length());
657 ESP_LOGD(__FILENAME__, "%s", payload.c_str());
658 } else {
659 ESP_LOGD(__FILENAME__, "Returned payload is empty");
660 }
661 if (http_code == 426) {
662 // HTTP status 426 is "Upgrade Required", which is the expected
663 // response for a websocket connection.
664 ESP_LOGD(__FILENAME__, "Attempting to connect to Signal K Websocket...");
665 server_detected_ = true;
666 token_test_success_ = true;
667 this->connect_ws(server_address, server_port);
668 } else if (http_code == 401) {
669 this->client_id_ = "";
670 this->send_access_request(server_address, server_port);
671 } else {
672 set_connection_state(SKWSConnectionState::kSKWSDisconnected);
673 }
674 } else {
675 ESP_LOGE(__FILENAME__, "HTTP request failed: %s", esp_err_to_name(err));
676 set_connection_state(SKWSConnectionState::kSKWSDisconnected);
677 }
678}

740 esp_http_client_cleanup(client);
741
742 // if we get a response we can't handle, try to reconnect later
743 if (err != ESP_OK || http_code != 202) {
744 ESP_LOGW(__FILENAME__, "Can't handle response %d to access request.",
745 http_code);
746 ESP_LOGD(__FILENAME__, "%s", payload.c_str());
747 set_connection_state(SKWSConnectionState::kSKWSDisconnected);
748 client_id_ = "";
749 return;
750 }
751
752 // http status code 202
753
754 deserializeJson(doc, payload.c_str());
755 String state = doc["state"];
756
757 if (state != "PENDING") {
758 ESP_LOGW(__FILENAME__, "Got unknown state: %s", state.c_str());
759 set_connection_state(SKWSConnectionState::kSKWSDisconnected);
760 client_id_ = "";
761 return;
762 }
763
764 String href = doc["href"];
765 polling_href_ = href;
766 save();
767
768 delay(5000);
769 this->poll_access_request(server_address, server_port, this->polling_href_);
770}

772void SKWSClient::poll_access_request(const String server_address,
773 const uint16_t server_port,
774 const String href) {
775 ESP_LOGD(__FILENAME__, "Polling SK Server for authentication token");
776
777 String protocol = ssl_enabled_ ? "https://" : "http://";
778 String url = protocol + server_address + ":" + server_port + href;
779
780 esp_http_client_config_t config = {};
781 config.url = url.c_str();
782 config.timeout_ms = 10000;
783#ifdef SENSESP_SSL_SUPPORT
784 if (ssl_enabled_) {
785 config.crt_bundle_attach = tofu_crt_bundle_attach;
786 config.skip_cert_common_name_check = true;
787 }
788#endif
789
790 esp_http_client_handle_t client = esp_http_client_init(&config);
791 if (client == nullptr) {
792 ESP_LOGE(__FILENAME__, "Failed to initialize HTTP client");
793 set_connection_state(SKWSConnectionState::kSKWSDisconnected);
794 return;
795 }
796
797 esp_err_t err = esp_http_client_perform(client);
798 int http_code = esp_http_client_get_status_code(client);
799 int content_length = esp_http_client_get_content_length(client);
800
801 String payload;
802 if (err == ESP_OK && content_length > 0 && content_length < 4096) {
803 char* buffer = new char[content_length + 1];
804 esp_http_client_read(client, buffer, content_length);
805 buffer[content_length] = '\0';
806 payload = String(buffer);
807 delete[] buffer;
808 }
809
810 esp_http_client_cleanup(client);
811
812 if (err != ESP_OK) {
813 ESP_LOGE(__FILENAME__, "HTTP request failed: %s", esp_err_to_name(err));
814 set_connection_state(SKWSConnectionState::kSKWSDisconnected);
815 return;
816 }
817
818 if (http_code == 200 || http_code == 202) {
819 JsonDocument doc;
820 auto error = deserializeJson(doc, payload.c_str());
821 if (error) {
822 ESP_LOGW(__FILENAME__, "WARNING: Could not deserialize http payload.");
823 ESP_LOGW(__FILENAME__, "DeserializationError: %s", error.c_str());
824 return;
825 }
826 String state = doc["state"];
827 ESP_LOGD(__FILENAME__, "%s", state.c_str());
828 if (state == "PENDING") {
829 set_connection_state(SKWSConnectionState::kSKWSDisconnected);
830 delay(5000);
831 return;
832 }
833 if (state == "COMPLETED") {
834 JsonObject access_req = doc["accessRequest"];
835 String permission = access_req["permission"];
836
837 polling_href_ = "";
838 save();
839
840 if (permission == "DENIED") {
841 ESP_LOGW(__FILENAME__, "Permission denied");
842 set_connection_state(SKWSConnectionState::kSKWSDisconnected);
843 return;
844 }
845
846 if (permission == "APPROVED") {
847 ESP_LOGI(__FILENAME__, "Permission granted");
848 String token = access_req["token"];
849 auth_token_ = token;
850 save();
851 this->connect_ws(server_address, server_port);
852 return;
853 }
854 }
855 } else {
856 if (http_code == 500) {
857 // this is probably the server barfing due to
858 // us polling a non-existing request. Just
859 // delete the polling href.
860 ESP_LOGD(__FILENAME__, "Got 500, probably a non-existing request.");
861 polling_href_ = "";
862 save();
863 set_connection_state(SKWSConnectionState::kSKWSDisconnected);
864 return;
865 }
866 // any other HTTP status code
867 ESP_LOGW(__FILENAME__,
868 "Can't handle response %d to pending access request.\n",
869 http_code);
870 set_connection_state(SKWSConnectionState::kSKWSDisconnected);
871 return;
872 }
873}

875void SKWSClient::connect_ws(const String& host, const uint16_t port) {
876 set_connection_state(SKWSConnectionState::kSKWSConnecting);
877
878 String protocol = ssl_enabled_ ? "wss" : "ws";
879 String path = "/signalk/v1/stream?subscribe=none";
880 String url = protocol + "://" + host + ":" + String(port) + path;
881
882 ESP_LOGD(__FILENAME__, "Connecting WebSocket to %s", url.c_str());
883
884 // Configure WebSocket client
885 esp_websocket_client_config_t config = {};
886 config.uri = url.c_str();
887 config.task_stack = 8192;
888 config.buffer_size = 1024;
889
890#ifdef SENSESP_SSL_SUPPORT
891 if (ssl_enabled_) {
892 // Use custom crt_bundle_attach to disable SSL verification
893 // This directly configures mbedTLS to skip certificate verification
894 config.crt_bundle_attach = tofu_crt_bundle_attach;
895 config.skip_cert_common_name_check = true;
896 }
897#endif
898
899 // Destroy any existing client
900 if (client_ != nullptr) {
901 esp_websocket_client_stop(client_);
902 esp_websocket_client_destroy(client_);
903 client_ = nullptr;
904 }
905
906 client_ = esp_websocket_client_init(&config);
907 if (client_ == nullptr) {
908 ESP_LOGE(__FILENAME__, "Failed to initialize WebSocket client");
909 set_connection_state(SKWSConnectionState::kSKWSDisconnected);
910 return;
911 }
912
913 // Register event handler
914 esp_websocket_register_events(client_, WEBSOCKET_EVENT_ANY,
915 websocket_event_handler, nullptr);
916
917 // Add authorization header if we have a token
918#ifdef SENSESP_SSL_SUPPORT
919 if (auth_token_ != NULL_AUTH_TOKEN) {
920 esp_websocket_client_append_header(client_, "Authorization",
921 (String("Bearer ") + auth_token_).c_str());
922 }
923#endif
924
925 // Start the client
926 esp_err_t err = esp_websocket_client_start(client_);
927 if (err != ESP_OK) {
928 ESP_LOGE(__FILENAME__, "Failed to start WebSocket client: %s",
929 esp_err_to_name(err));
930 esp_websocket_client_destroy(client_);
931 client_ = nullptr;
932 set_connection_state(SKWSConnectionState::kSKWSDisconnected);
933 return;
934 }
935
936 ESP_LOGD(__FILENAME__, "WebSocket client started, waiting for connection...");
937}

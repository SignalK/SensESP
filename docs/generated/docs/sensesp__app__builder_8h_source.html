323 const SensESPAppBuilder* enable_wifi_watchdog() {
324 // create the wifi disconnect watchdog
325 app_->system_status_controller_
326 ->connect_to(new Debounce<SystemStatus>(
327 3 * 60 * 1000 // 180 s = 180000 ms = 3 minutes
328 ))
329 ->connect_to(new LambdaConsumer<SystemStatus>([](SystemStatus input) {
330 ESP_LOGD(__FILENAME__, "Got system status: %d", (int)input);
331 if (input == SystemStatus::kWifiDisconnected ||
332 input == SystemStatus::kWifiNoAP) {
333 ESP_LOGW(__FILENAME__,
334 "Unable to connect to wifi for too long; restarting.");
335 event_loop()->onDelay(1000, []() { ESP.restart(); });
336 }
337 }));
338
339 return this;
340 }

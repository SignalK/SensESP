24void ExecuteWebSocketTask(void* parameter) {
25 elapsedMillis connect_loop_elapsed = 0;
26 elapsedMillis delta_loop_elapsed = 0;
27 elapsedMillis ws_client_loop_elapsed = 0;
28
29 ws_client->connect();
30
31 while (true) {
32 if (connect_loop_elapsed > 2000) {
33 connect_loop_elapsed = 0;
34 ws_client->connect();
35 }
36 if (delta_loop_elapsed > 5) {
37 delta_loop_elapsed = 0;
38 ws_client->send_delta();
39 }
40 if (ws_client_loop_elapsed > 20) {
41 ws_client_loop_elapsed = 0;
42 ws_client->loop();
43 }
44 delay(1);
45 }

57void webSocketClientEvent(WStype_t type, uint8_t* payload, size_t length) {

303void WSClient::on_receive_put(DynamicJsonDocument& message) {
304 // Process PUT requests...
305 JsonArray puts = message["put"];
306 size_t response_count = 0;
307 for (size_t i = 0; i < puts.size(); i++) {
308 JsonObject value = puts[i];
309 const char* path = value["path"];
310 String strVal = value["value"].as<String>();
311
312 SKListener::take_semaphore();
313 const std::vector<SKPutListener*>& listeners =
314 SKPutListener::get_listeners();
315 for (size_t i = 0; i < listeners.size(); i++) {
316 SKPutListener* listener = listeners[i];
317 if (listener->get_sk_path().equals(path)) {
318 take_received_updates_semaphore();
319 received_updates_.push_back(value);
320 release_received_updates_semaphore();
321 response_count++;
322 }
323 }
324 SKListener::release_semaphore();
325
326 // Send back a request response...
327 DynamicJsonDocument put_response(512);
328 put_response["requestId"] = message["requestId"];
329 if (response_count == puts.size()) {
330 // We found a response for every PUT request
331 put_response["state"] = "COMPLETED";
332 put_response["statusCode"] = 200;
333 } else {
334 // One or more requests did not have a matching path
335 put_response["state"] = "FAILED";
336 put_response["statusCode"] = 405;
337 }
338 String response_text;
339 serializeJson(put_response, response_text);
340 this->client_.sendTXT(response_text);
341 }

371void WSClient::connect() {
372 if (get_connection_state() != WSConnectionState::kWSDisconnected) {
373 return;
374 }
375
376 if (!WiFi.isConnected() && WiFi.getMode() != WIFI_MODE_AP) {
377 debugI(
378 "WiFi is disconnected. SignalK client connection will connect when "
379 "WiFi is connected.");
380 return;
381 }
382
383 debugI("Initiating websocket connection with server...");
384
385 set_connection_state(WSConnectionState::kWSAuthorizing);
386 String server_address = this->server_address_;
387 uint16_t server_port = this->server_port_;
388
389 if (this->server_address_.isEmpty()) {
390 if (!get_mdns_service(server_address, server_port)) {
391 debugE("No Signal K server found in network when using mDNS service!");
392 } else {
393 debugI("Signal K server has been found at address %s:%d by mDNS.",
394 server_address.c_str(), server_port);
395 }
396 }
397
398 if (!server_address.isEmpty() && server_port > 0) {
399 debugD("Websocket is connecting to Signal K server on address %s:%d",
400 server_address.c_str(), server_port);
401 } else {
402 // host and port not defined - wait for mDNS
403 set_connection_state(WSConnectionState::kWSDisconnected);
404 return;
405 }
406
407 if (this->polling_href_ != "") {
408 // existing pending request
409 this->poll_access_request(server_address, server_port, this->polling_href_);
410 return;
411 }
412
413 if (this->auth_token_ == NULL_AUTH_TOKEN) {
414 // initiate HTTP authentication
415 debugD("No prior authorization token present.");
416 this->send_access_request(server_address, server_port);
417 return;
418 }
419
420 // Test the validity of the authorization token
421 this->test_token(server_address, server_port);
422}

466void WSClient::send_access_request(const String server_address,
467 const uint16_t server_port) {
468 debugD("Preparing a new access request");
469 if (client_id_ == "") {
470 // generate a client ID
471 client_id_ = generate_uuid4();
472 save_configuration();
473 }
474
475 // create a new access request
476 DynamicJsonDocument doc(1024);
477 doc["clientId"] = client_id_;
478 doc["description"] =
479 String("SensESP device: ") + SensESPBaseApp::get_hostname();
480 doc["permissions"] = kRequestPermission;
481 String json_req = "";
482 serializeJson(doc, json_req);
483
484 debugD("Access request: %s", json_req.c_str());
485
486 HTTPClient http;
487
488 String url = String("http://") + server_address + ":" + server_port +
489 "/signalk/v1/access/requests";
490 debugD("Access request url: %s", url.c_str());
491 http.begin(wifi_client_, url);
492 http.addHeader("Content-Type", "application/json");
493 int httpCode = http.POST(json_req);
494 String payload = http.getString();
495 http.end();
496
497 // if we get a response we can't handle, try to reconnect later
498 if (httpCode != 202) {
499 debugW("Can't handle response %d to access request.", httpCode);
500 debugD("%s", payload.c_str());
501 set_connection_state(WSConnectionState::kWSDisconnected);
502 client_id_ = "";
503 return;
504 }
505
506 // http status code 202
507
508 deserializeJson(doc, payload.c_str());
509 String state = doc["state"];
510
511 if (state != "PENDING") {
512 debugW("Got unknown state: %s", state.c_str());
513 set_connection_state(WSConnectionState::kWSDisconnected);
514 client_id_ = "";
515 return;
516 }
517
518 String href = doc["href"];
519 polling_href_ = href;
520 save_configuration();
521
522 delay(5000);
523 this->poll_access_request(server_address, server_port, this->polling_href_);
524}

526void WSClient::poll_access_request(const String server_address,
527 const uint16_t server_port,
528 const String href) {
529 debugD("Polling SK Server for authentication token");
530
531 HTTPClient http;
532
533 String url = String("http://") + server_address + ":" + server_port + href;
534 http.begin(wifi_client_, url);
535 int httpCode = http.GET();
536 if (httpCode == 200 or httpCode == 202) {
537 String payload = http.getString();
538 http.end();
539 DynamicJsonDocument doc(1024);
540 auto error = deserializeJson(doc, payload.c_str());
541 if (error) {
542 debugW("WARNING: Could not deserialize http payload.");
543 debugW("DeserializationError: %s", error.c_str());
544 return; // TODO: return at this point, or keep going?
545 }
546 String state = doc["state"];
547 debugD("%s", state.c_str());
548 if (state == "PENDING") {
549 delay(5000);
550 this->poll_access_request(server_address, server_port, href);
551 return;
552 } else if (state == "COMPLETED") {
553 JsonObject access_req = doc["accessRequest"];
554 String permission =
555 access_req["permission"]; // TODO: like this in ArdJson 6? String
556 // permission =
557 // resp["accessRequest"]["permission"];
558 polling_href_ = "";
559 save_configuration();
560
561 if (permission == "DENIED") {
562 debugW("Permission denied");
563 set_connection_state(WSConnectionState::kWSDisconnected);
564 return;
565 } else if (permission == "APPROVED") {
566 debugI("Permission granted");
567 String token = access_req["token"];
568 auth_token_ = token;
569 save_configuration();
570 this->connect_ws(server_address, server_port);
571 return;
572 }
573 }
574 } else {
575 http.end();
576 if (httpCode == 500) {
577 // this is probably the server barfing due to
578 // us polling a non-existing request. Just
579 // delete the polling href.
580 debugD("Got 500, probably a non-existing request.");
581 polling_href_ = "";
582 save_configuration();
583 set_connection_state(WSConnectionState::kWSDisconnected);
584 return;
585 }
586 // any other HTTP status code
587 debugW("Can't handle response %d to pending access request.\n", httpCode);
588 set_connection_state(WSConnectionState::kWSDisconnected);
589 return;
590 }

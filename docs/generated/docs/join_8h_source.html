17template <int N>
18class JoinBase {
19 public:
20 JoinBase(long max_age = 0) : max_age_{max_age} {
21 for (int i = 0; i < N; i++) {
22 age_[i] = max_age;
23 }
24 }
25
26 protected:
27 elapsedMillis age_[N];
28 long max_age_;
29
30 virtual void emit_tuple() = 0;
31
32 void check_emit() {
33 for (int i = 0; i < N; i++) {
34 if (max_age_ != 0 && age_[i] > max_age_) {
35 return;
36 }
37 }
38 emit_tuple();
39 }
40};

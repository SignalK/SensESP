328 void SensorNXP_FXOS8700_FXAS21002::setupSensors(void) {
329  //Sets ranges for sensors, when defaults are not optimal.
330  //Note that if the ranges are changed, the scaling of the output values
331  //may need adjusting. For example, the accelerometer is operating
332  //by default at +/-2g sensitivity (= 0.244 mg/LSB), and the accelerations
333  //are converted to m/s^2. The conversion factor needs to scale
334  //proportionally to the sensitivity.
335  //Register locations and settings are from the datasheets for the
336  //FXAS21002 and FXOS8700:
337  //https://www.nxp.com/docs/en/data-sheet/FXAS21002.pdf
338  //https://www.nxp.com/docs/en/data-sheet/FXOS8700CQ.pdf
339 
340  //set gyro data rate to correspond with sampling rate, so
341  //internal LPF (low pass filter) will be suitable freq
342  uint8_t odr_bits;
343  if (sampling_rate_hz_ > 400.0) {
344  odr_bits = 0b000;
345  }else if(sampling_rate_hz_ > 200.0) {
346  odr_bits = 0b001;
347  }else if(sampling_rate_hz_ > 100.0) {
348  odr_bits = 0b010;
349  }else if(sampling_rate_hz_ > 50.0) {
350  odr_bits = 0b011;
351  }else if(sampling_rate_hz_ > 25.0) {
352  odr_bits = 0b100;
353  }else if(sampling_rate_hz_ > 12.5) {
354  odr_bits = 0b101;
355  }else {
356  odr_bits = 0b110;
357  }
358  odr_bits = odr_bits << FXAS_CR1_ODR_BITS_LOC; //shift left to bit posns 4:2
359 
360  byte ctl_reg1 = readByte(FXAS21002C_ADDRESS,GYRO_REGISTER_CTRL_REG1); //get reg value
361  ctl_reg1 &= ~(FXAS_CR1_ACTIVE_BITMASK | FXAS_CR1_READY_BITMASK); //zero out the Active & Ready bits
362  writeByte(FXAS21002C_ADDRESS,GYRO_REGISTER_CTRL_REG1, ctl_reg1); //put gyro in standby
363  ctl_reg1 = (ctl_reg1 & (~(FXAS_CR1_ODR_BITMASK | FXAS_CR1_ACTIVE_BITMASK))) | odr_bits | FXAS_CR1_ACTIVE_BITMASK;
364  writeByte(FXAS21002C_ADDRESS,GYRO_REGISTER_CTRL_REG1, ctl_reg1 ); //back to Active mode
365 
366 } // end setupSensors()

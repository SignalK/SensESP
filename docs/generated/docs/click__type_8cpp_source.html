32void ClickType::on_button_press() {
33 ESP_LOGD(
34 __FILENAME__,
35 "ClickType received PRESS on click count %d (millis: %ld, last release: "
36 "%ld ms ago)",
37 click_count_, millis(), (long)release_duration_);
38
39 if (click_count_ == 0) {
40 // This is a new, isolated "click" that we have not yet processed.
41 click_count_++;
42 press_duration_ = 0;
43 } else {
44 // One or more presses is already in progress...
45
46 if (press_duration_ > ultra_long_click_delay_) {
47 // The button down is the second one reported in a row without a button
48 // release and the press is now long enough to qualify as an "ultra long
49 // click"
50 on_ultra_long_click();
51 } else if (release_duration_ <= double_click_interval_) {
52 // This is the start of a second click to come in prior to the expiration
53 // of the double_click_interval. Remove any "SingleClick" report that may
54 // have been queued up....
55 if (delayed_click_report_ != NULL) {
56 delayed_click_report_->remove();
57 delayed_click_report_ = NULL;
58 ESP_LOGD(__FILENAME__,
59 "ClickType press is double click. Removed queued SingleClick "
60 "report");
61 }
62 click_count_++;
63 }
64 }
65
66 this->emit(ClickTypes::ButtonPress);
67}

69void ClickType::on_button_release() {
70 ESP_LOGD(__FILENAME__,
71 "ClickType received UNPRESS for click count %d (millis: %ld, press "
72 "duration: %ld ms)",
73 click_count_, millis(), (long)press_duration_);
74
75 if (click_count_ > 0) {
76 // This is the "release" of a click we are tracking...
77 if (press_duration_ >= this->ultra_long_click_delay_) {
78 on_ultra_long_click();
79 this->on_click_completed();
80 } else if (press_duration_ >= this->long_click_delay_) {
81 ESP_LOGD(
82 __FILENAME__,
83 "ClickType detected LongSingleClick (millis: %ld, press duration %ld "
84 "ms)",
85 millis(), (long)press_duration_);
86 this->emitDelayed(ClickTypes::LongSingleClick);
87 this->on_click_completed();
88 } else if (this->click_count_ > 1) {
89 // We have just ended a double click. Sent it immediately...
90 ESP_LOGD(
91 __FILENAME__,
92 "ClickType detected DoubleClick (millis: %ld, press duration %ld ms)",
93 millis(), (long)press_duration_);
94 this->emitDelayed(ClickTypes::DoubleClick);
95 this->on_click_completed();
96 } else {
97 // This is the end of a potential single click. Queue up the send of a
98 // SingleClick report, but delay it in case another click comes in prior
99 // to the double_click_interval, which would turn this click into a
100 // DoubleClick
101 unsigned long time_of_event = millis();
102 long pd = (long)press_duration_;
103 delayed_click_report_ = ReactESP::app->onDelay(
104 double_click_interval_ + 20, [this, pd, time_of_event]() {
105 ESP_LOGD(
106 __FILENAME__,
107 "ClickType detected SingleClick (millis: %ld, queue time: %ld, "
108 "press duration %ld ms)",
109 millis(), time_of_event, pd);
110 this->emit(ClickTypes::SingleClick);
111 this->on_click_completed();
112 });
113 }
114
115 release_duration_ = 0;
116 this->emit(ClickTypes::ButtonRelease);
117
118 } else {
119 // A press release with no initial press should happen only when
120 // an UltraLongClick has already been sent, or the producer
121 // is feeding us weird values...
122 release_duration_ = 0;
123 ESP_LOGW(__FILENAME__,
124 "ClickType detected UNPRESS with no pending PRESS (millis=%ld)",
125 millis());
126 }
127}

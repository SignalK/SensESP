527 void NMEAParser::state_in_term(char c) {
528  switch (c) {
529  case ',':
530  case '*':
531  if (cur_offset<INPUT_BUFFER_LENGTH) {
532  // split terms with 0 to help further processing
533  buffer[cur_offset++] = 0;
534  } else {
535  current_state = &NMEAParser::state_start;
536  break;
537  }

547  } else {

549  }

554  buffer[cur_offset++] = 0;
555  current_state = &NMEAParser::state_start;
556  break;
557  default:

565 void NMEAParser::state_in_checksum(char c) {
566  switch (c) {
567  case ',':
568  case '*':

570  current_state = &NMEAParser::state_start;
571  case '\r':
572  case '\n':

574  buffer[cur_offset++] = 0;

576  current_state = &NMEAParser::state_start;
577  return;
578  }
579  // call the relevant sentence parser

582  } else {
583  sentence_parsers[buffer]->parse(buffer, term_offsets, cur_term+1,
584  sentence_parsers);
585  }
586  current_state = &NMEAParser::state_start;
587  break;
588  default:
589  // read term characters
590  buffer[cur_offset++] = c;
